    <script>
        let balance = parseFloat(localStorage.getItem('minBal')) || 0.00;
        let historyArr = JSON.parse(localStorage.getItem('minHist')) || [];

        function updateUI() {
            document.getElementById('userBalance').innerText = balance.toFixed(2);
            const list = document.getElementById('historyList');
            list.innerHTML = historyArr.length ? '' : '<div style="color:#555; text-align:center;">Операций нет</div>';
            historyArr.forEach(item => {
                const div = document.createElement('div'); div.className = 'history-item';
                div.innerHTML = `<span>${item.name}</span><span style="color:${item.color}">${item.val}</span>`;
                list.appendChild(div);
            });
            localStorage.setItem('minBal', balance); localStorage.setItem('minHist', JSON.stringify(historyArr));
        }

        updateUI();
        function openModal(id) { document.getElementById(id).style.display = 'block'; document.getElementById('overlay').style.display = 'block'; }
        function closeAll() { document.querySelectorAll('.modal').forEach(m => m.style.display = 'none'); document.getElementById('overlay').style.display = 'none'; }
        function copy(text) { navigator.clipboard.writeText(text); alert('Адрес скопирован!'); }
        function updateDepositText() { document.getElementById('finalAmount').innerText = (parseFloat(document.getElementById('sumInput').value) || 0).toFixed(2) + ' TON'; }
        function goToPay(p) { document.getElementById('sumInput').value = p; updateDepositText(); closeAll(); openModal('depositModal'); }
        function confirmDeposit() {
            let v = parseFloat(document.getElementById('sumInput').value);
            if(v > 0) { historyArr.unshift({name: "Пополнение (проверка)", val: "+" + v.toFixed(2) + " TON", color: "#f1c40f"}); alert("Заявка отправлена!"); updateUI(); closeAll(); }
        }
        function processWithdraw() {
            let s = parseFloat(document.getElementById('withdrawSum').value);
            if(s > balance) return alert("Недостаточно средств!");
            if(s >= 10) { balance -= s; historyArr.unshift({name: "Вывод", val: "-" + s.toFixed(2) + " TON", color: "#e74c3c"}); updateUI(); alert("Заявка на вывод принята!"); closeAll(); } else { alert("Минимум 10 TON!"); }
        }
    </script>
